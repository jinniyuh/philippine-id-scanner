How to Use:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Include PHPMailer in your PHP files:

   require_once 'includes/PHPMailer/PHPMailer.php';
   require_once 'includes/PHPMailer/SMTP.php';
   require_once 'includes/PHPMailer/Exception.php';

   use PHPMailer\PHPMailer\PHPMailer;
   use PHPMailer\PHPMailer\SMTP;
   use PHPMailer\PHPMailer\Exception;

2. Create a new PHPMailer instance:

   $mail = new PHPMailer(true);

3. Configure SMTP (for Hostinger):

   $mail->isSMTP();
   $mail->Host       = 'smtp.hostinger.com';
   $mail->SMTPAuth   = true;
   $mail->Username   = 'bagovet_info@bccbsis.com';
   $mail->Password   = 'Y^k*/[ElK4c';
   $mail->SMTPSecure = PHPMailer::ENCRYPTION_SMTPS;
   $mail->Port       = 465;

4. Send email:

   $mail->setFrom('bagovet_info@bccbsis.com', 'Bago Vet');
   $mail->addAddress('recipient@example.com');
   $mail->Subject = 'Test Email';
   $mail->Body = 'This is a test email';
   $mail->send();

Troubleshooting:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Issue: "Class 'PHPMailer' not found"
Fix: Check require_once paths are correct

Issue: "SMTP connect() failed"
Fix: Verify SMTP settings and port 465 not blocked

Issue: "Could not instantiate mail function"
Fix: Check PHP mail() function is enabled

Issue: OpenSSL error
Fix: Ensure OpenSSL extension is loaded in php.ini
